40  /*
41  Covert Channel, that uses 3 bits of the TCP Header, that are not used, to hide a message.
42  To use this class, call 'sendMessage' with your message as a string. Received packets are used to call 'receiveMessage'.
43  The decoded message is given to the callback function 'm_output'.
44  */
45  class TcpHeaderCovertChannel : public CovertChannel {
46  public:
47  /*
48  Constructor
49  output is a function pointer that is called, when a complete message arrived. Its parameter is this message.
50  send is a function pointer that is called to send a TcpPacket via the socket.
51  getPacket is a function pointer, that has to return a valid TcpPacket, that is used to insert the data.
52  */
53  TcpHeaderCovertChannel(function<void(string)> output, function<void(TcpPacket)> send, function<TcpPacket(void)> getPacket)
54  : CovertChannel(),
55  m_remaining_packets(0),
56  m_output(output),
57  m_send(send),
58  m_getPacket(getPacket) {};
59 
60  // SendMessage sends a message of type string through the Tcp Header Covert Channel.
61  void sendMessage(string message);
62 
63  /*
64  This function is called, when a new packet arrives at the socket.
65  It collects them and returns the message via the callback function "m_output".
66  */
67  void receiveMessage(GenericPacket& packet);
68 
69  // No available arguments - empty function
70  void setArguments(string arguments) {};

79  string protocol() const;

81  // Returns the used port (8080)
82  short port() const;

84  private:

86  vector<bitset<3> > encodeMessageWithLength(string message);
87 
88  // modifyTcpPacket sets the reserved bits of packet to the value of data.

111
