31 typedef boost::bimap<char, string> morse_map;

34 namespace whisper_library {
35  /*
36  MorseCoder converts a message into time intervals using morse and decodes morse back to a message.
37  To do so, call 'encodeMessage' with the message or 'decodeMessage' with time intervals.
38  Use 'checkString' to check if the given message contains characters, that are not supported by morse.
39  */
40  class MorseCoder {
41  public:
42  /*
43  Constructor
44  delay_short is used to encode a short signal (in milliseconds)
45  delay_long is used to encode a long signal (in milliseconds)
46  delay_letter is used to encode the end of a letter (in milliseconds)
47  delay_space is used to encode space between words (in milliseconds)
48  */
49  MorseCoder(unsigned int delay_short, unsigned int delay_long, unsigned int delay_letter, unsigned int delay_space);
50 
51  /*
52  encodeMessage takes a message as a string and converts it into a sequence of delays using morse.
53  Unsupported characters are encoded as '#'.
54  */
55  vector<unsigned int> encodeMessage(string message);
56  /*
57  decodeMessage takes a sequence of delays representing morse and converts them back to a message.
58  The delays have to be exactly the ones set in the constructor.
59  */
60  string decodeMessage(vector<unsigned int> delays);
61 
62  /*
63  Checks the given string for characters, that are not supported by morse.
64  If none were found, the returned vector is empty. Otherwise it contains unsupported characters.
65  */
66  vector<char> checkString(string message);
67 
68  private:
69  /*
70  Encodes a single character to morse.
71  The encoded character is returned as a vector of bool in which 0 (false) means short and 1 (true) means long.
72  */
73  vector<bool> encodeLetter(char letter);
74 
75  /*
76  Decodes morse for a single character
77  The passed vector of bool has to contain the encoded letter in which 0 (false) means short and 1 (true) means long.
78  */
79  char decodeLetter(vector<bool> morse_code);
80 
81  // Represents a short signal (in milliseconds)
82  unsigned int m_delay_short;
83  // Represents a long signal (in milliseconds)
84  unsigned int m_delay_long;
85  // Represents a short pause (in milliseconds)
86  unsigned int m_delay_letter;
87  // Represents a long pause (in milliseconds)
88  unsigned int m_delay_space;
89 
90  // Contains the mapping between characters and morse.
91  morse_map m_morse_map;
92  };
93 }
